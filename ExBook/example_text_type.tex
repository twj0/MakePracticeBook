        \begin{solution}

    \begin{bbox}
        \qitem[][]   设$f(x)=2x+\sqrt{x^{2}+2x+1}$，$g(x)=
        \begin{cases}
        x+2, & x\geqslant0, \\
        x-1, & x<0,  
        \end{cases}$，则$g[f(x)]= \blankline$.
        \begin{solution}
            解：$x^2+2x+1=(x+1)^2$，所以$f(x)=2x+|x+1|$
            
            当$x\geq -1$时，$f(x)=2x+(x+1)=3x+1$,
            当$x<-1$时，$f(x)=2x-(x+1)=x-1$.
            
            因为$f(x)$是连续函数，且$f(-1)=2(-1)+0=-2<0$,
            所以当$x<-1$时，$f(x)<0$，此时$g[f(x)]=f(x)-1=x-2$,
            当$x\geq -1$时，$f(x)=3x+1\geq -2$，此时$g[f(x)]=f(x)+2=3x+3$.
            
            答案：$g[f(x)]=
            \begin{cases}
            3x+3, & x\geq -1 \\
            x-2, & x<-1
            \end{cases}$
        \end{solution}
    \end{bbox}
